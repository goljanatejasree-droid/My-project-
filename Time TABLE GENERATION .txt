    <script>
        // --- DATA STORAGE ---
        let faculty = ['Librarian', 'Sports Instructor'];
        let rooms = ['Library', 'Sports Ground'];
        let sections = [];
        let facultyConstraints = [];
        let nextId = 0;

        // --- CONSTANTS ---
        const DAYS = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
        const PERIODS = 8;
        const LUNCH_PERIOD = 3; // 4th period, index 3

        // --- INITIALIZATION ---
        document.addEventListener('DOMContentLoaded', () => {
            populateTimeDropdowns();
            renderFaculty();
            renderRooms();
        });

        function populateTimeDropdowns() {
            const dayDropdown = document.getElementById('constraint-day');
            const periodDropdown = document.getElementById('constraint-period');
            dayDropdown.innerHTML = '<option value="">Select Day</option>' + DAYS.map((day, i) => `<option value="${i}">${day}</option>`).join('');
            periodDropdown.innerHTML = '<option value="">Select Period</option>' + Array.from({length: PERIODS}, (_, i) => i === LUNCH_PERIOD ? '' : `<option value="${i}">Period ${i+1}</option>`).join('');
        }

        // --- DOM MANIPULATION FUNCTIONS ---

        function addFaculty() {
            const input = document.getElementById('faculty-name');
            const name = input.value.trim();
            if (name && !faculty.find(f => f.toLowerCase() === name.toLowerCase())) {
                faculty.push(name);
                faculty.sort();
                input.value = '';
                renderFaculty();
                updateAllFacultyDropdowns();
            }
        }

        function renderFaculty() {
            document.getElementById('faculty-list').innerHTML = faculty
                .filter(f => !['Librarian', 'Sports Instructor'].includes(f))
                .map(f => `<li class="flex justify-between items-center bg-slate-100 p-2 rounded-md"><span>${f}</span><button class="text-red-500 text-xs font-semibold" onclick="removeFaculty('${f}')">REMOVE</button></li>`).join('');
        }

        function removeFaculty(name) {
            faculty = faculty.filter(f => f !== name);
            facultyConstraints = facultyConstraints.filter(c => c.faculty !== name);
            renderFaculty();
            updateAllFacultyDropdowns();
            renderFacultyConstraints();
        }
        
        function addRoom() {
            const input = document.getElementById('room-name');
            const name = input.value.trim();
            if (name && !rooms.find(r => r.toLowerCase() === name.toLowerCase())) {
                rooms.push(name);
                rooms.sort();
                input.value = '';
                renderRooms();
                updateAllRoomDropdowns();
            }
        }
        
        function renderRooms() {
             document.getElementById('room-list').innerHTML = rooms
                .filter(r => !['Library', 'Sports Ground'].includes(r))
                .map(r => `<li class="flex justify-between items-center bg-slate-100 p-2 rounded-md"><span>${r}</span><button class="text-red-500 text-xs font-semibold" onclick="removeRoom('${r}')">REMOVE</button></li>`).join('');
        }

        function removeRoom(name) {
            rooms = rooms.filter(r => r !== name);
            renderRooms();
            updateAllRoomDropdowns();
        }

        function addSection() {
            const input = document.getElementById('section-name');
            const name = input.value.trim();
            if (name && !sections.find(s => s.name === name)) {
                sections.push({ id: nextId++, name, subjects: [] });
                input.value = '';
                renderSections();
            }
        }
        
        function toggleLabDuration(sectionId) {
            const isLabCheckbox = document.getElementById(`is-lab-${sectionId}`);
            const durationContainer = document.getElementById(`lab-duration-container-${sectionId}`);
            const hoursInput = document.getElementById(`subject-hours-${sectionId}`);
            
            if (isLabCheckbox.checked) {
                durationContainer.style.display = 'block';
                hoursInput.disabled = true;
                hoursInput.value = ''; 
            } else {
                durationContainer.style.display = 'none';
                hoursInput.disabled = false;
            }
        }

        function renderSections() {
            const container = document.getElementById('sections-container');
            container.innerHTML = sections.map(section => `
                <div class="border p-4 rounded-lg bg-white" id="section-${section.id}">
                    <div class="flex justify-between items-center mb-4">
                        <h3 class="text-lg font-semibold">${section.name}</h3>
                        <button class="text-red-500 font-medium" onclick="removeSection(${section.id})">Remove Section</button>
                    </div>
                    <div id="subjects-${section.id}" class="space-y-2 mb-4">${renderSubjectsForSection(section)}</div>
                    <div class="bg-blue-50 p-3 rounded-lg">
                        <h4 class="font-medium mb-2 text-blue-800">Add Subject/Lab to ${section.name}</h4>
                        <div class="grid grid-cols-1 sm:grid-cols-2 gap-2">
                            <input type="text" id="subject-name-${section.id}" placeholder="Subject Name" class="p-2 border rounded-md">
                            <input type="number" id="subject-hours-${section.id}" placeholder="Hours/Week" class="p-2 border rounded-md" min="1" value="1">
                            <select id="subject-faculty-${section.id}" class="p-2 border rounded-md faculty-dropdown"><option value="">Select Faculty</option></select>
                            <select id="subject-room-${section.id}" class="p-2 border rounded-md room-dropdown"><option value="">Select Room</option></select>
                            <div class="col-span-2 flex items-center gap-4">
                               <input type="checkbox" id="is-lab-${section.id}" class="h-4 w-4 rounded border-gray-300 text-indigo-600 focus:ring-indigo-500" onchange="toggleLabDuration(${section.id})">
                               <label for="is-lab-${section.id}">This is a Lab</label>
                            </div>
                            <div id="lab-duration-container-${section.id}" class="col-span-2" style="display:none;">
                                <label for="lab-duration-${section.id}" class="text-sm font-medium">Lab Duration (consecutive hours)</label>
                                <select id="lab-duration-${section.id}" class="w-full p-2 border rounded-md">
                                    <option value="2">2 hours</option>
                                    <option value="3">3 hours</option>
                                </select>
                            </div>
                            <button onclick="addSubject(${section.id})" class="sm:col-span-2 bg-blue-500 text-white p-2 rounded-md hover:bg-blue-600 transition">Add Subject</button>
                        </div>
                    </div>
                </div>
            `).join('');
            updateAllFacultyDropdowns();
            updateAllRoomDropdowns();
        }
        
        function removeSection(id) {
            sections = sections.filter(s => s.id !== id);
            renderSections();
        }
        
        function addSubject(sectionId) {
            const section = sections.find(s => s.id === sectionId);
            const name = document.getElementById(`subject-name-${sectionId}`).value.trim();
            const facultyName = document.getElementById(`subject-faculty-${sectionId}`).value;
            const roomName = document.getElementById(`subject-room-${sectionId}`).value;
            const isLab = document.getElementById(`is-lab-${sectionId}`).checked;
            
            let hours;
            if (isLab) {
                hours = parseInt(document.getElementById(`lab-duration-${sectionId}`).value, 10);
            } else {
                hours = parseInt(document.getElementById(`subject-hours-${sectionId}`).value, 10);
            }

            if (name && hours > 0 && facultyName && roomName) {
                section.subjects.push({ name, hours, faculty: facultyName, room: roomName, isLab });
                document.getElementById(`subjects-${sectionId}`).innerHTML = renderSubjectsForSection(section);
                ['subject-name', 'subject-hours', 'subject-faculty', 'subject-room'].forEach(id => document.getElementById(`${id}-${sectionId}`).value = '');
                document.getElementById(`is-lab-${sectionId}`).checked = false;
                toggleLabDuration(sectionId);
            } else {
                alert("Please fill all subject details, including faculty and room.");
            }
        }

        function renderSubjectsForSection(section) {
            return section.subjects.map((sub, index) => `
                <div class="grid grid-cols-3 items-center text-sm bg-slate-50 p-2 rounded-md gap-2">
                    <span class="col-span-2"><strong>${sub.name}</strong> (${sub.hours}h) - <em>${sub.faculty}</em> [${sub.room}] ${sub.isLab ? '<span class="text-xs bg-green-200 text-green-800 p-1 rounded">LAB</span>' : ''}</span>
                    <button class="text-red-500 text-xs justify-self-end font-semibold" onclick="removeSubject(${section.id}, ${index})">REMOVE</button>
                </div>
            `).join('');
        }

        function removeSubject(sectionId, subjectIndex) {
            const section = sections.find(s => s.id === sectionId);
            section.subjects.splice(subjectIndex, 1);
            document.getElementById(`subjects-${sectionId}`).innerHTML = renderSubjectsForSection(section);
        }
        
        function addFacultyConstraint() {
            const facultyName = document.getElementById('constraint-faculty').value;
            const day = parseInt(document.getElementById('constraint-day').value);
            const period = parseInt(document.getElementById('constraint-period').value);

            if (facultyName && day >= 0 && period >= 0) {
                const constraint = { faculty: facultyName, day, period };
                if (!facultyConstraints.some(c => c.faculty === facultyName && c.day === day && c.period === period)) {
                    facultyConstraints.push(constraint);
                    renderFacultyConstraints();
                }
            } else {
                alert('Please select a faculty, day, and period.');
            }
        }

        function renderFacultyConstraints() {
            const list = document.getElementById('faculty-constraint-list');
            list.innerHTML = facultyConstraints.map((c, i) => `
                <li class="flex justify-between items-center bg-amber-100 p-1 rounded-md">
                    <span>${c.faculty} is busy on ${DAYS[c.day]} at Period ${c.period + 1}</span>
                    <button class="text-red-500 text-xs font-semibold" onclick="removeFacultyConstraint(${i})">X</button>
                </li>
            `).join('');
        }

        function removeFacultyConstraint(index) {
            facultyConstraints.splice(index, 1);
            renderFacultyConstraints();
        }

        function updateAllFacultyDropdowns() {
            const dropdowns = document.querySelectorAll('.faculty-dropdown');
            const options = faculty.filter(f => !['Librarian', 'Sports Instructor'].includes(f)).map(f => `<option value="${f}">${f}</option>`).join('');
            dropdowns.forEach(dropdown => {
                const selectedValue = dropdown.value;
                dropdown.innerHTML = '<option value="">Select Faculty</option>' + options;
                dropdown.value = selectedValue;
            });
        }
        
        function updateAllRoomDropdowns() {
            const dropdowns = document.querySelectorAll('.room-dropdown');
            const options = rooms.filter(r => !['Library', 'Sports Ground'].includes(r)).map(r => `<option value="${r}">${r}</option>`).join('');
            dropdowns.forEach(dropdown => {
                const selectedValue = dropdown.value;
                dropdown.innerHTML = '<option value="">Select Room</option>' + options;
                dropdown.value = selectedValue;
            });
        }

        // --- TIMETABLE GENERATION LOGIC ---

        function generateTimetable() {
            if (sections.length === 0) {
                alert("Please add at least one section.");
                return;
